import RPi.GPIO as GPIO
import time
import matplotlib.pyplot as plt

class R2R_ADC:
    def __init__(self, dynamic_range, compare_time=0.01, verbose=False):
        self.dynamic_range = dynamic_range
        self.verbose = verbose
        self.compare_time = compare_time
        
        self.bits_gpio = [26, 20, 19, 16, 13, 12, 25, 11]
        self.comp_gpio = 21
        
        GPIO.setmode(GPIO.BCM)
        GPIO.setup(self.bits_gpio, GPIO.OUT, initial=0)
        GPIO.setup(self.comp_gpio, GPIO.IN)
    
    def number_to_dac(self, number):
        """Подает число number на вход ЦАП"""
        for i, pin in enumerate(self.bits_gpio):
            bit_value = (number >> i) & 1
            GPIO.output(pin, bit_value)
    
    def sequential_counting_adc(self):
        """Последовательный счетный АЦП"""
        max_number = (1 << len(self.bits_gpio)) - 1
        
        for number in range(max_number + 1):
            self.number_to_dac(number)
            time.sleep(self.compare_time)
            
            comparator_state = GPIO.input(self.comp_gpio)
            
            if comparator_state == 1:
                return number
        
        return max_number
    
    def get_sc_voltage(self):
        """Возвращает измеренное напряжение в Вольтах"""
        digital_value = self.sequential_counting_adc()
        max_digital_value = (1 << len(self.bits_gpio)) - 1
        voltage = (digital_value / max_digital_value) * self.dynamic_range
        return voltage
    
    def __del__(self):
        """Деструктор - очистка GPIO"""
        GPIO.cleanup()

def plot_voltage_vs_time(time_values, voltage_values, max_voltage):
    """Строит график зависимости напряжения от времени"""
    plt.figure(figsize=(10, 6))
    plt.plot(time_values, voltage_values)
    plt.title('Зависимость напряжения от времени')
    plt.xlabel('Время, с')
    plt.ylabel('Напряжение, В')
    plt.ylim(0, max_voltage * 1.1)
    plt.grid(True)
    plt.show()

def plot_sampling_period_hist(time_values):
    """Строит распределение количества измерений по их продолжительности"""
    sampling_periods = []
    
    # Заполняем список разницами между соседними моментами времени
    for i in range(1, len(time_values)):
        period = time_values[i] - time_values[i-1]
        sampling_periods.append(period)
    
    plt.figure(figsize=(10, 6))
    plt.hist(sampling_periods, bins=20, edgecolor='black')
    plt.title('Распределение периодов измерений')
    plt.xlabel('Период измерения, с')
    plt.ylabel('Количество измерений')
    plt.xlim(0, 0.06)
    plt.grid(True)
    plt.show()

# Создайте объект класса R2R_ADC
adc = R2R_ADC(dynamic_range=3.3)

# Создайте два списка:
voltage_values = []
time_values = []
duration = 3.0  # Продолжительность измерений

try:
    # Сохраните момент начала эксперимента
    start_time = time.time()
    
    # Пока разница между текущим временем и начальным моментом меньше продолжительности эксперимента:
    while (time.time() - start_time) < duration:
        # Добавляйте в список измеренные значения напряжений
        voltage = adc.get_sc_voltage()
        voltage_values.append(voltage)
        
        # Добавляйте в список моменты времени
        current_time = time.time() - start_time
        time_values.append(current_time)

finally:
    # Вызовите деструктор класса
    adc.__del__()

# Построение графиков
if voltage_values and time_values:
    plot_voltage_vs_time(time_values, voltage_values, 3.3)
    plot_sampling_period_hist(time_values)
